 12 public class BinaryTree<K, V> implements BinaryTreeADT<K, V> {
 13     
 14     /**
 15      * The instance variable representing the left node.
 16      * Will be empty if the node doesn't exist.
 17      */
 18     private BinaryTree<K, V> left;
 19     /**
 20      * The instance variable representing the right node.
 21      * Will be empty if the node doesn't exist
 22      */
 23     private BinaryTree<K, V> right;
 24     /**
 25      * The key instance variable
 26      */
 27     private K key;
 28     /**
 29      * The value instance variable
 30      */
 31     private V value;
 32     
 33     /**
 34      *  Creates a new BinaryTree object with all instance variables
 35      *  set to null
 36      */
 37     public BinaryTree() {
 38         key = null;
 39     }
 40 
 41     /**
 42      * Getter function for the value instance variable
 43      * @return this.value
 44      */
 45     @Override
 46     public V value() {
 47         return value;
 48     }
 49         
 50     /**
 51      * Setter function for the value instance variable
 52      * @param value The new value for this.value
 53      */
 54     @Override
 55     public void setValue(V value) {
 56         this.value = value;
 57     }
 58 
 59     /**
 60      * Returns if this current node is empty. 
 61      * An empty node is the conceptual equivalent of
 62      * a non existent node.
 63      * An empty node entails having a null key,
 64      * which is the default value when the constructor is called.
 65      * @return is this node empty
 66      */
 67     @Override
 68     public boolean empty() {
 69         return key == null;
 70     }
 71     
 72     /**
 73      * Gets the greater of two integers
 74      * @param a
 75      * @param b
 76      * @return the greater of a and b 
 77      */
 78     private int max(int a, int b) { return a > b ? a : b; }
 79     
 80     /**
 81      * Recursively finds the height of the current tree.
 82      * @return the height of the current tree 
 83      */
 84     @Override
 85     public int height() {
 86         if (empty()) {
 87             return 0;
 88         } else {
 89             return 1 + max(left.height(), right.height());
 90         }
 91     }
 92 
 93     /**
 94      * Recursively prints the elements in the tree in preorder (i.e 
 95      * visit, left.preorder(), right.preorder() .
 96      */
 97     @Override
 98     public void preorder() {
 99         if (!empty()) {
100             visit();
101             left.preorder();
102             right.preorder();
103         }
104     }
105     
106     /**
107      * Prints the value and key of the current node.
108      */
109     public void visit() {
110         if (value == null) {
111             System.out.println("[" + key + ":" + "---" + "]" );
112         } else {
113             System.out.println("[" + key + ":" + value + "]" );
114         }
115     }
116     
117     /**
118      * Recursively prints the elements in the tree in inorder (i.e 
119      * left.inorder(), visit, right.inorder() .
120      */
121     @Override
122     public void inorder() {
123         if (!empty()) {
124             left.inorder();
125             visit();
126             right.inorder();
127         }
128     }

130     /**
131      * Recursively prints the elements in the tree in postorder (i.e

133      */
134     @Override
135     public void postorder() {
136         if (!empty()) {
137             left.postorder();
138             right.postorder();
139             visit();
140         }
141     }

143     /**
144      * Recursively adds a given key and value to the node specified by dir
145      * @param key The key to be added
146      * @param value The value to be added
147      * @param dir The directory to place the new node in, specified by a string
148      *              containing L or R depending on the path to be traversed
149      * @throws BinaryTreeCreationException Throws this exception if the path is invalid
150      */
151     @Override
152     public void addLR(K key, V value, String dir) throws BinaryTreeCreationException {
153         try {
154             if (dir.equals("")) {
155                 sprout(key, value);
156             } else if (dir.charAt(0) == 'L' || dir.charAt(0) == 'l') {
157                 left.addLR(key, value, dir.substring(1));
158             } else if (dir.charAt(0) == 'R' || dir.charAt(0) == 'r') {
159                 right.addLR(key, value, dir.substring(1));
160             } else {
161                 throw new BinaryTreeCreationException("Fug");
162             }
163         } catch(BinaryTreeCreationException e) {
164             throw e;
165         }
166     }
167 
168     /**
169      * Recursively adds a given key and value non determinantly to
170      * @param key   The key to be added
171      * @param value The value to be added
172      */
173     @Override
174     public void addND(K key, V value) {
175         if (empty()) {
176             sprout(key, value);
177         } else if (Math.random() < 0.5) {
178             left.addND(key, value);
179         } else {
180             right.addND(key, value);
181         }
182     }

184     /**
185      * Recursively adds a given key and value to the tree as a binary search tree
186      * @param key the key to be added
187      * @param value the value to be added
188      * @throws BinaryTreeCreationException This is thrown if the value is equal to another node 
189      */
190     @Override
191     public void addST(K key, V value) throws BinaryTreeCreationException {
192         if (empty()) {
193             sprout(key, value);
194         } else {
195             Comparable cey  = (Comparable) key;
196             Comparable t    = (Comparable) this.key;
197             if (cey.compareTo(t) < 0) {
198                 left.addST(key, value);
199             } else if (cey.compareTo(t) > 0) {
200                 right.addST(key, value);
201             } else {
202                 throw new BinaryTreeCreationException("fug");
203             }
204         }
205     }

207     /**
208      * Gets the value with the given key
209      * @param key they key used to find the value
210      * @return The value with key
211      */
212     @Override
213     public V get(K key) {
214         return find(key).value();
215     }
216 
217     /**
218      * Recursively finds the value with the given key
219      * @param key the key to search for
220      * @return The tree with the given key
221      */
222     @Override
223     public BinaryTreeADT<K, V> find(K key) {
224         if (empty()) {
225             return null;
226         } else if (this.key == key) {
227             return this;
228         } else {
229             BinaryTreeADT<K, V> l = left.find(key);
230             if (l != null) return l;
231             BinaryTreeADT<K, V> r = right.find(key);
232             if (l != null) return r;
233             return null;
234         }
235     }
236     
237     /**
238      * Checks if a node with the given key exists in the current ree
239      * @param key the node to check for presence
240      * @return If the current tree contains a node with the given key
241      */
242     @Override
243     public boolean member(K key) {
244         return find(key) != null;
245     }
246 
247     /**
248      * Makes the current node (which should be empty) into a 
249      * node with the given key and value
250      * @param key   The key 
251      * @param value The value
252      */
253     @Override
254     public void sprout(K key, V value) {
255         this.key = key;
256         this.value = value;
257         this.left = new BinaryTree<>();
258         this.right = new BinaryTree<>();
259     }
