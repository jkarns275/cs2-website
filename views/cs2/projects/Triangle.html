21         this.a = a;
22         this.b = b;
23         this.c = c;
24     }
25 
26     @Override
27     public double perimeter() {
28         return a + b + c;
29     }
30 
31     @Override
32     public double area() {
33         double p = (a + b + c) / 2;
34         return Math.sqrt(p*(p-a)*(p-b)*(p-c));
35     }
36     
37     @Override
38     public String toString() {
39         return "<Triangle with A=" + a + " B=" + b + " C=" + c + ">";
40     }
