 9  *

12 public class Name implements NameADT {
13     
14     private String firstName, lastName;
15     
16     public Name(String first, String last) {
17         firstName   = first;
18         lastName    = last;
19     }
20 
21     @Override
22     public String first() {
23         return firstName;
24     }
25 
26     @Override
27     public String last() {
28         return lastName;
29     }
30 
31     @Override
32     public String natural() {
33         return firstName + " " + lastName;
34     }
35 
36     @Override
37     public String formal() {
38         return lastName + ", " + firstName;
39     }
40 
41     @Override
42     public boolean check(String fi, String li) {
43         return fi.equals(firstName.substring(0, 1)) && li.equals(lastName.substring(0, 1));
44     }
45     
46     @Override
47     public String toString() {
48         return natural();
49     }
50     
51     public String initials() {
52         return "" + firstName.charAt(0) + lastName.charAt(0);
53     }
54 }
