 8 import java.awt.BorderLayout;
 9 import java.awt.Color;
10 import java.awt.Container;
11 import java.awt.event.ActionEvent;
12 import java.awt.event.ActionListener;
13 import javax.swing.JButton;
14 import javax.swing.JFrame;
15 import javax.swing.JPanel;
16 
17 /**
18  *
19  * @author jkarn
20  */
21 public class KFrame6 extends JFrame implements ActionListener {
22         
23         private JButton blueButton;
24         private JButton redButton;
25         private JButton greenButton;
26         private JButton yellowButton;
27         private JPanel reflector;
28         
29         public KFrame6(String title) {
30             super(title);
31             setSize(500, 300);
32             setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
33             addComponents(getContentPane());
34             addListeners();
35             setVisible(true);
36         }
37         
38         private void addListeners() {
39             blueButton.addActionListener(this);
40             redButton.addActionListener(this);
41             greenButton.addActionListener(this);
42             yellowButton.addActionListener(this);
43         }
44         
45         @Override
46         public void actionPerformed(ActionEvent event) {
47             String command = event.getActionCommand();
48             if (command.equals("Red"))
49                 reflector.setBackground(Color.red);
50             else if (command.equals("Blue"))
51                 reflector.setBackground(Color.blue);
52             else if (command.equals("Green"))
53                 reflector.setBackground(Color.green);
54             else if (command.equals("Yellow"))
55                 reflector.setBackground(Color.yellow);
56             
57         }
58         
59         private void addComponents(Container contentPane) {
60             blueButton = new JButton("Blue");
61             redButton = new JButton("Red");
62             greenButton = new JButton("Green");
63             yellowButton = new JButton("Yellow");
64             reflector = new JPanel();
65             contentPane.setLayout(new BorderLayout());
66             contentPane.add(blueButton, BorderLayout.NORTH);
67             contentPane.add(redButton, BorderLayout.SOUTH);
68             contentPane.add(greenButton, BorderLayout.EAST);
69             contentPane.add(yellowButton, BorderLayout.WEST);
70             contentPane.add(reflector, BorderLayout.CENTER);
71         }
72         
73     }
