17 public class ArrayListPlay {
18     
19     public static void main(String[] args) {
20         ArrayList songs = createSongList();
21         display("The songs ... ", songs);
22         ArrayList actors = createActorList();
23         display("The actors ... ", actors);
24         ArrayList triangles = createTriangleList();
25         display("The triangles ... ", triangles);
26         
27         displayTitles(songs);
28         displayInitials(actors);
29         displayAreas(triangles);
30     }
31 
32     private static ArrayList createSongList() {
33         return new ArrayList<>(Arrays.asList(
34         "The Calendar",
35         "Always",
36         "Death Of A Bachelor",
37         "No More Bad Days",
38         "Pink + White",
39         "Somethin' Stupid",
40         "There's A Good Reason These Tables Are Numbered Honey, You Just Haven't Thought Of It Yet",
41         "Tear In My Heart"
42         ));
43     }
44 
45     private static ArrayList createActorList() {
46         return new ArrayList<>(Arrays.asList(
47         new Name("Danny", "Devito"),
48         new Name("Leonardo", "DiCaprio"),
49         new Name("Tom", "Hanks"),
50         new Name("Nicolas", "Cage"),
51         new Name("Morgan", "Freeman"),
52         new Name("Michael", "Caine"),
53         new Name("Christian", "Bale")
54         ));
55     }
56 
57     private static ArrayList createTriangleList() {
58         ArrayList<Triangle> triangleList = new ArrayList<>();
59         for (int i = 0 ; i < 6 ; i++) {
60             double a = Math.random() * 100;
61             double b = Math.random() * 100;
62             double c = Math.sqrt(a * a + b * b - 2 * 5 * 7 * Math.cos(Math.random() * 150 + .1));
63             System.out.println(c);
64             triangleList.add(new Triangle(a, b, c));
65         }
66         return triangleList;
67     }    
68 
69     private static void display(String str, ArrayList list) {
70         System.out.println(str);
71         for (Object o : list)
72             System.out.println(o);
73     }
74     
75     private static void displayTitles(ArrayList<String> songs) {
76         System.out.println("The song title lengths ...");
77         for (String song : songs)
78             System.out.println(song.length());
79     }
80     
81     private static void displayInitials(ArrayList<Name> names) {
82         System.out.println("The initials of the names ...");
83         for (Name name : names)
84             System.out.println(name.initials());
85     }
86     
87     private static void displayAreas(ArrayList<Triangle> triangles) {
88         System.out.println("The areas of the triangles ...");
89         for (Triangle triangle : triangles)
90             System.out.println(triangle.area());
91     }
