19 public class FrequencyCounterTrees {
20     public static void main(String[] args) {
21         BinaryTree<String, Integer> frequencies = new BinaryTree<>();
22         Scanner input = new Scanner(System.in);
23         System.out.print(">>> Filename? ");
24         String filename = input.nextLine();
25
26         try {
27             Scanner fileInput = new Scanner(new File(filename));
28             fileInput.forEachRemaining((key) -> {
29                 if (frequencies.find(key) != null) {
30                     frequencies.find(key).setValue(frequencies.get(key) + 1);
31                 } else
32                     try {
33                         frequencies.addST(key, 1);
34                     } catch (BinaryTreeCreationException ex) {
35                         Logger.getLogger(FrequencyCounterTrees.class.getName()).log(Level.SEVERE, null, ex);
36                     }
37             });
38         } catch (Exception e) {
39             System.out.println("Error :(");
40             e.printStackTrace();
41         }
42         frequencies.inorder();
43     }
44 }
