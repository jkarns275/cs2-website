 6 package hashing;
 7
 8 import java.io.File;
 9 import java.util.HashMap;
10 import java.util.Map;
11 import java.util.Scanner;
12
13 /**
14  *
15  * @author jkarn
16  */
17 public class FrequencyCounterHashing {
18     static Map<String, Integer> frequencies = new HashMap<String, Integer>();
19     static Scanner input = new Scanner(System.in);
20
21     public static void main(String[] args) {
22         establishFrequencyCounts();
23         dump();
24         interpret();
25     }
26     public static void establishFrequencyCounts() {
27         System.out.println(">>> Filename? ");
28         String filename = input.nextLine();
29
30         try {
31             Scanner fileInput = new Scanner(new File(filename));
32             while (fileInput.hasNext()) {
33                 String key = fileInput.next().toLowerCase();
34                 if (frequencies.containsKey(key))
35                     frequencies.put(key, frequencies.get(key) + 1);
36                 else
37                     frequencies.put(key, 1);
38             }
39         } catch (Exception e) {
40             System.out.println("Error :(");
41             e.printStackTrace();
42         }
43     }
44
45     public static void dump() {
46         frequencies.forEach((k, v) -> {
47             System.out.println(k + " -> " + v);
48         });
49     }
50
51     public static void interpret() {
52          while (true) {
53             System.out.print(">>> ");
54             String line = input.next().toLowerCase();
55             if (line.equals("exit")) {
56                 System.out.println("Goodbye!\n");
57                 return;
58             } else if (line.equals("print")) {
59                 frequencies.forEach((k, v) -> {
60                     System.out.println(k + " -> " + v);
61                 });
62             } else if (line.equals("count")) {
63                 System.out.print(">>> Which word? ");
64                 String key = input.next();
65                 System.out.println(key + " -> " + frequencies.get(key));
66             } else {
67                 System.out.println(">>> Unrecognized Command");
68
69             }
70         }
71     }
72 }
