 4 package review;

 6 import java.util.Scanner;
 7
 8 public class LongestWord {
 9
10     static String getSentence() {
11         System.out.printf("Please enter a sentence: ");
12         Scanner s = new Scanner(System.in);
13         return s.nextLine();
14     }
15
16     static String getLongestWord(String sentence, int start, int end, int longestStart, int longestEnd) {
17         if (end >= sentence.length()) return sentence.substring(longestStart, longestEnd);
18         if (sentence.charAt(end) == ' ')
19             if (longestEnd - longestStart < end - start)
20                 return getLongestWord(sentence, end + 1, end + 1, start, end);
21             else
22                 return getLongestWord(sentence, end + 1, end + 1, longestStart, longestEnd);
23         else
24             return getLongestWord(sentence, start, end + 1, longestStart, longestEnd);
25     }
26
27     public static void main(String[] args) {
28         System.out.printf("The longest word is \"%s\"\n", getLongestWord(getSentence() + " ", 0, 0, 0, 0));
29     }
