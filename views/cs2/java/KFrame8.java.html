116     public void actionPerformed(ActionEvent event) {
117         String command = event.getActionCommand().toUpperCase();
118         if (event.getSource() == input) {
119             input.setText("");
120         }
121         if (command.equals("CLEAR")) {
122             output.setText("");
123             output.setBackground(Color.WHITE);
124         } else if (command.equals("LINE")) {
125             output.append(lineToString(randomLine()) + "\n");
126         } else if (command.equals("WORD")) {
127              this.output.append(wordBank.nextWord() + "\n");
128         } else if (command.equals("QUOTE")) {
129             this.output.append(quoteBank.nextQuote() + "\n");
130         } else if (command.equals("COLOR")) {
131             this.output.setBackground(new Color(random.nextInt()));
132         } else if (command.equals("POINT")) {
133             this.output.append(pointToString(randomPoint()) + "\n");
134         }
135     }

137     private String lineToString(Point p) {
138         return "Line { length = " + p.x + "\t; angle = " + p.y + "deg }"; 
139     }
140     
141     private String pointToString(Point p) {
142         return "Line { x = " + p.x + "\t; y = " + p.y + " }"; 
143     }
144     
145     private Point randomPoint() {
146         return new Point(random.nextInt(501), random.nextInt(601));
147     }

149     /**

154         return new Point(random.nextInt(601), random.nextInt(360));
155     }
