  6 package linearlists;

  8 import java.util.LinkedList;
  9 import java.util.List;
 10 import java.util.Random;
 11 import java.util.stream.Collectors;
 12 import java.util.stream.DoubleStream;
 13 import java.util.stream.IntStream;
 14 import java.util.stream.Stream;
 15 
 16 /**
 17  *
 18  * @author wua
 19  */
 20 public class LinkedListPlay {
 21     public static void main(String[] args) {
 22         LinkedList animals = createAnimalList();
 23         displayAnimals("\nThe animal names ... ", animals);
 24         
 25         LinkedList reals = randomList(77);
 26         displayReals("\nThe real numbers ... ", reals);
 27         
 28         LinkedList lengths = lengths(animals);
 29         displayLengths("\nThe lengths of the animal names ... ", lengths);
 30 
 31         LinkedList reverseAnimals = reverse(animals);
 32         displayAnimals("\nThe animal names in reverse order ... ", reverseAnimals);
 33         
 34         double minimum = minimum(reals);
 35         System.out.println("\nThe minimum number = " + minimum);
 36         displayReals("\nThe real numbers ... ", reals);
 37         
 38         LinkedList orderedList = sort(reals);
 39         displayReals("\nThe real numbers in ascending order ... ", orderedList);
 40         displayReals("\nThe real numbers ... ", reals);
 41     }
 42 
 43     private static LinkedList createAnimalList() {
 44         return Stream.of("lobster", "walrus", "bat", "jackalope", "iquana").collect(Collectors.toCollection(LinkedList::new));
 45     }
 46 
 47     private static void displayObjects(String message, List list) {
 48         System.out.println(message);
 49         System.out.print("< ");
 50         for (Object o : list) System.out.print(o + " ");
 51         System.out.println(" >");
 52     }
 53     
 54     private static void displayAnimals(String message, LinkedList<String> animals) {
 55         displayObjects(message, animals);
 56     }
 57 
 58     private static LinkedList<Double> randomList(int i) {
 59         return IntStream.rangeClosed(0, i - 1).mapToDouble(x -> Math.random() * 100).boxed().collect(Collectors.toCollection(LinkedList::new));
 60     }
 61 
 62     private static void displayReals(String message, LinkedList<Double> reals) {
 63         displayObjects(message, reals);
 64     }
 65 
 66     private static void displayLengths(String message, LinkedList lengths) {
 67         displayObjects(message, lengths);
 68     }
 69 
 70     private static LinkedList<Integer> lengths(LinkedList<String> animals) {
 71         return IntStream.rangeClosed(0, animals.size() - 1).map(x -> animals.get(x).length()).boxed().collect(Collectors.toCollection(LinkedList::new));
 72     }
 73 
 74     private static LinkedList reverse(LinkedList list) {
 75         LinkedList newList = new LinkedList(list);
 76         IntStream.rangeClosed(0, newList.size() / 2).forEach(i -> {
 77             newList.set(i, list.get(list.size() - 1 - i));
 78             newList.set(newList.size() - 1 - i, list.get(i));
 79         });
 80         return newList;
 81     }
 82 
 83     private static double minimum(LinkedList<Double> reals) {
 84         double min = Double.POSITIVE_INFINITY;
 85         for (Double d : reals) min = min > d ? d : min;
 86         return min;
 87     }
 88     
 89     private static int indexOfMinimum(LinkedList<Double> reals, int startInd) {
 90         int indOfMin = startInd;
 91         for (int d = startInd ; d < reals.size() ; d++)  {
 92             indOfMin = reals.get(indOfMin) > reals.get(d) ? d : indOfMin;
 93         }
 94         return indOfMin;
 95     }
 96     
 97     private static void swap(LinkedList<Double> reals, int a, int b) {
 98         Double temp = reals.get(a);
 99         reals.set(a, reals.get(b));
100         reals.set(b, temp);
101     }
102 
103     private static LinkedList sort(LinkedList<Double> reals) {
104         LinkedList<Double> copy = new LinkedList<>(reals);
105         LinkedList<Double> sorted = new LinkedList<>();
106         for (int i = 0 ; i < copy.size() ; i++) {
107             int indexOfMin = indexOfMinimum(copy, i);
108             sorted.add(copy.get(indexOfMin));
109             swap(copy, i, indexOfMin);
110         }
111         return sorted;
112     }
