37     private class KFrame extends JFrame implements ActionListener {
38
39         private JButton blueButton;
40         private JButton redButton;
41         private JButton greenButton;
42         private JButton yellowButton;
43         private JPanel reflector;
44
45         public KFrame(String title) {
46             super(title);
47             setSize(500, 300);
48             setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
49             addComponents(getContentPane());
50             addListeners();
51             setVisible(true);
52         }
53
54         private void addListeners() {
55             blueButton.addActionListener(this);
56             redButton.addActionListener(this);
57             greenButton.addActionListener(this);
58             yellowButton.addActionListener(this);
59         }
60
61         @Override
62         public void actionPerformed(ActionEvent event) {
63             String command = event.getActionCommand();
64             if (command.equals("Red"))
65                 reflector.setBackground(Color.red);
66             else if (command.equals("Blue"))
67                 reflector.setBackground(Color.blue);
68             else if (command.equals("Green"))
69                 reflector.setBackground(Color.green);
70             else if (command.equals("Yellow"))
71                 reflector.setBackground(Color.yellow);
72
73         }
74
75         private void addComponents(Container contentPane) {
76             blueButton = new JButton("Blue");
77             redButton = new JButton("Red");
78             greenButton = new JButton("Green");
79             yellowButton = new JButton("Yellow");
80             reflector = new JPanel();
81             contentPane.setLayout(new BorderLayout());
82             contentPane.add(blueButton, BorderLayout.NORTH);
83             contentPane.add(redButton, BorderLayout.SOUTH);
84             contentPane.add(greenButton, BorderLayout.EAST);
85             contentPane.add(yellowButton, BorderLayout.WEST);
86             contentPane.add(reflector, BorderLayout.CENTER);
87         }
88
89     }
