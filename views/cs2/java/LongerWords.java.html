 1 package review;
 2
 3 import java.util.Scanner;
 4
 5 /*
 6  *  This program prompts the user for a sentence, and then finds the average
 7  * word length in that sentence. It will then print all words with a higher than
 8  * average length.
 9  */
10 public class LongerWords {
11
12     static String[] getSentence() {
13         System.out.printf("Please enter a sentance: ");
14         Scanner s = new Scanner(System.in);
15         return s.nextLine().split(" ");
16     }
17
18     static float getAverageLength(String[] words) {
19         float average = 0.f;
20         for (String word : words) average += word.length();
21         average /= words.length;
22         return average;
23     }
24
25     public static void main(String[] args) {
26         String[] words = getSentence();
27         float average = getAverageLength(words);
28         System.out.printf("The longer words: \n");
29         for (String word : words)
30             if ((float) word.length() > average)
31                 System.out.printf("%s\n", word);
32     }
33
34 }
